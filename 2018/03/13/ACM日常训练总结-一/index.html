Encoding

传送门

Problem Description

Given a string containing only ‘A’ - ‘Z’, we could encode it using the following method:

  1. Each sub-string containing k same characters should be encoded to “kX” where “X” is the only character in this sub-string.
  2. If the length of the sub-string is 1, ‘1’ should be ignored.

Input

The first line contains an integer N (1 <= N <= 100) which indicates the number of test cases. The next N lines contain N strings. Each string consists of only ‘A’ - ‘Z’ and the length is less than 10000.

Output

For each test case, output the encoded string in a line

Sample Input

2 ABC ABBCCC

Sample Output

ABC A2B3C

这个题目坑的地方主要在与你要理解清楚它是把每个子串做加密，而不是最后把相同的字母统计然后输出，我一开始以为的是统计整个序列中每个字符的个数输出，所以写的代码如下：

#include<iostream>
#include<cstdio>
#include<string.h>
using namespace std;

int main(){
  int N;
  char ch;
  int flag=1;
  scanf("%d", &N);
  while(N--){
    int a[30]={0};
    if(flag) {getchar(); flag=0;}
    while(true){
    ch=getchar();
    if(ch == '\n') break;
    else{
      a[ch-65] += 1;
    }
    }
    for(int i = 0 ; i< 26 ; i++){
      if(a[i]==1) cout << char(i+65);
      else if(a[i] > 1) cout << a[i] << char(i+65);
    }
    cout << endl;
  }

  return 0;
}

因为HDU上面没有多的测试样例可以看，所以一开始错了好几次，后来才发现是Each-substring，所以用C++中的string类来做就比较简单了：

#include <iostream>  
#include <algorithm>  
#include <cstdio>  
#include <string>  
#include <cstring>  
using namespace std;  
int main()  
{  
    int t;  
    string s;  
    cin>>t;  
    getchar();  
    while(t--)  
    {  
        cin>>s;  
        int len=s.size();  
        int ans=1;///用ans来记录字符出现的次数  
        int temp=s[0];///用temp来存储第一个字符  
        for(int i=1; i<=len; i++)///将temp与后边的字符进行比较，相同的ans++  
        {  
            if(s[i]!=temp)  
            {  
                if(ans==1)///如果次数为1，忽略1，直接输出  
                {  
                    cout<<s[i-1];  
                    temp=s[i];  
                }  
                else  
                {  
                    cout<<ans<<s[i-1];  
                    temp=s[i];  
                    ans=1;  
                }  
            }  
            else  
                ans++;  
        }  
        cout<<endl;  
    }  
    return 0;  
}

Least Common Multiple

Problem Description
The least common multiple (LCM) of a set of positive integers is the smallest positive integer which is divisible by all the numbers in the set. For example, the LCM of 5, 7 and 15 is 105.

Input
Input will consist of multiple problem instances. The first line of the input will contain a single integer indicating the number of problem instances. Each instance will consist of a single line of the form m n1 n2 n3 … nm where m is the number of integers in the set and n1 … nm are the integers. All integers will be positive and lie within the range of a 32-bit integer.

Output
For each problem instance, output a single line containing the corresponding LCM. All results will lie in the range of a 32-bit integer.

Sample Input 

2 
3 5 7 15 
6 4 10296 936 1287 792 1

Sample Output 

105 
10296

乍一看我们要求很多数的总共最大公约数，可能一开始会慌了神，如果没有想到循环求两个数的最大公约数可能会用穷举的办法，因为最大公约数必定是最大的那个数的倍数，所以这样写代码也是可以过的：

#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;

int main(){
  int n,t;
  scanf("%d", &n);
  while(n--){
    scanf("%d", &t);
    int a[100001]={0};
    for(int i = 0 ; i < t ; i++)
      scanf("%d", &a[i]);
    sort(a, a+t);
    int max = a[t-1];
    int p;
    for(p=max;;p+=max){
      int opp=0;
      for(int i = 0 ; i < t ; i++){
        if(p%a[i]!=0){
          opp++;
          break;
        }
      }
      if(!opp) break;
    }
    printf("%d\n", p);
  }
  return 0;
}

但是我们可以在输入的时候做一些改变，然后循环的去求解最大公约数，这样算法复杂度就降低了很多：

#include<stdio.h>  

int gcd(int a, int b) {  
    return b?gcd(b, a%b):a;  
}  

int main(void)  
{  
    int T, n, x, temp;  
    scanf("%d", &T);  

    while (T--) {  
    scanf("%d %d", &n, &temp);  
        for (int i = 1; i < n; ++i) {  
            scanf("%d", &x);  
        //lcm
          temp = temp / gcd(temp, x) * x;  
        }  
        printf("%d\n", temp);  
    }  

    return 0;  
}

A+B Problem II

传送门

这个大整数加法如果直接加肯定会越界，所以我们先转换成字符串，然后转换回来每一位进行加法运算，结果放进整型数组，然后逐个输出，这样就可以解决问题了：

#include<iostream>
#include<cstdio>
#include<string.h>
using namespace std;

int main(){
  int N;
  scanf("%d", &N);
  for(int i = 1 ; i <= N ; i++){
    char s1[1001],s2[1002];
    int a[1002]={0},b[1001]={0},c[1001]={0};
    int len1,len2,len3;
    int max;
    int p,q;
    int m,n;
    scanf("%s %s", s1, s2);
    len1 = strlen(s1);
    len2 = strlen(s2);
    if(len1 > len2) max = len1;
    else max = len2;
    for(p = 0 , q = len1-1 ; p < len1 ; p++, q--)
      a[p] = s1[q] - 48;
    for(p = 0 , q = len2 - 1 ; p < len2 ; p++ , q--)
      b[p] = s2[q] - 48;
    for(m = 0, n = 0 ; n < max ; n++){
      c[n] = (a[n] + b[n] + m)%10;
      m = (a[n] + b[n] + m)/10;
    }
    printf("Case %d:\n%s + %s = ", i, s1, s2);
    if(!m){
      for(len3=max-1 ; len3 >= 0 ; len3--)
        printf("%d", c[len3]);
    }else{
      c[max] = 1;
      for(len3=max ; len3 >= 0 ; len3--)
        printf("%d", c[len3]);
    }
    if(i==N){
      printf("\n");
    }else{
      printf("\n\n");
    }
  }
  return 0;
}

Max Sum

传送门

这个其实可以用动态规划的思想去解决，其实我们每一次判断的本质就是：

sum = max(sum + t, t)

然后注意一下起始点和终止点的位置就好了，水题鉴定完毕。

#include <stdio.h>  
int main()  
{  
    int z,n,max,sum;  
    int a,b,A,B,t;  
    scanf("%d",&z);  
    for(int k=1;k<=z;k++)  
    {  
        scanf("%d",&n);  
        sum = max = -1001;  
        for(int i=1;i<=n;i++)  
        {  
            scanf("%d",&t);  
            if(sum+t < t)  
                sum = t , a = b = i;      //a、b记录当前连续子序列的起始、结束位置  
            else  
                sum += t , ++b;  
            if(max < sum)  
                max = sum , A = a , B = b;  
        }  
        printf("Case %d:\n%d %d %d\n",k,max,A,B);  
        if(k-z) puts("");  
    }  
    return 0;  
}

免费馅饼

传送门

转换成数塔就很好做了

#include <stdio.h>  
#include <string.h>  
int dp[100005][15];  
int max(int a,int b,int c)  
{  
    int max=a;  
    if(b>max)  
        max=b;  
    if(c>max)  
        max=c;  
    return max;  
}  
int main()  
{  
    int n,i,j;  
    while(~scanf("%d",&n))  
    {  
        if(n==0)  
            break;  
        int maxT=0;  
        memset(dp,0,sizeof(dp));  
        for(i=0;i<n;i++)  
        {  
            int x,t;  
            scanf("%d %d",&x,&t);  
              //避免做边界判断，所以从1开始，那么max取出来的dp[i][0]就是0
            dp[t][x+1]++;  
            if(t>maxT)  
                maxT=t;  
        }
        for( i = maxT - 1 ; i >= 0 ; i--)  
        {  
            for(j = 1 ; j <= 11 ; j++)  
            {  
                dp[i][j]+=max(dp[i+1][j-1],dp[i+1][j+1],dp[i+1][j]);  
            }  
        }  
        printf("%d\n",dp[0][5]);  
    }  
    return 0;  
}

Let the Ballon Rise

传送门

这个可以直接通过对于STL的掌握来完成，用到upper_bound和lower_bound两个函数：

#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;

int main(){
  int n;
  string s[1010];
  while(scanf("%d", &n) && n){
    for(int i = 0 ; i < n ; i++)
      cin >> s[i];
    sort(s, s+n);
    int cnt = 0;
    int ans = -1;
    int maxl = -1;
    for(int i = 0 ; i < n ;){
      cnt = upper_bound(s, s+n, s[i]) - lower_bound(s, s+n, s[i]);
      if(cnt > maxl){
        maxl = cnt;
        ans = i;
      }
      i += cnt;
    }
    cout << s[ans] << endl;
  }
  return 0;
}
