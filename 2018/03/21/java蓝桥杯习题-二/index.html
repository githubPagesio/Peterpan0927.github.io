蓝桥杯中java组的注意事项

  • 类名称必须采用public class Main方式命名

  • 整数之间的运算为整数。

  • int的取值范围为（-2147483648~2147483647），占用4个字节（-2的31次方到2的31次方-1），一个字节占8位，也就是32位。

  • 所以注意题目给出的参数范围是否在int的取值范围内，或者当参数运算后的值是否在int的取值范围。

  • 浮点数运算可能存在误差。在进行浮点数比较时，应考虑到浮点数误差。

  • 浮点数进行运算、比较是相当危险的！
    例如：

    System.out.println(0.2+0.1==0.3);
//运行结果：false12

    所以浮点数的比较不能用==。
    道理：

    浮点数在计算机内部是采用二进制（B Binary）的形式来表示的。
0.1=1/2  0.01=1/4  0.001=1/8...
十进制的0.3在二进制中是无限循环小数，从而会造成舍入
0.3=0.299999999997or0.30000000000004
12345

    处理手段：
    1.扩大倍数，转化为整数。
    2.任意精度的浮点数：java.math包。

包子凑数

这个题的就是用到了扩展欧几里得的变形，如果满足所有数的最大公约数不为1则有无穷个，否则都是有限个。如果是有限个就用完全背包统计即可，下面是java实现的代码：

import java.util.Scanner;

class Baozi{
  public static int gcd(int a, int b){
    return b==0 ? a : gcd(b, a%b);
  }
  
  public static int N = 10010;
  public static boolean[]  bk = new boolean[N];
  
  public static void main(String[] args){
    Scanner x = new Scanner(System.in);
    int m = x.nextInt();
    int[] number = new int[m];
    for(int i = 0 ; i < m ; i++){
        number[i] = x.nextInt();
    }
    int g = number[0];
    for(int i = 0 ; i < m ; i++){
      g = gcd(g, number[i]);
    }
    if(g != 1){
      System.out.println("INF");
    }else{
      bk[0] = true;
      for(int i = 0 ; i < m ; i ++)
        {
            for(int j = 0 ; j + number[i] < N ; j ++)
                if(bk[j]) bk[j+number[i]]= true;
        }
        int count = 0;
        for(int i = N-1 ; i >= 0 ; i --){
            if(bk[i] == false) count++;
        }
        System.out.println(count);
    }
  }
}

日期问题

这个题目考主要是严谨，要对于年月日的边界条件判断，根据闰年来调整2月的天数等。

import java.util.ArrayList;
import java.util.Collections;
import java.util.Scanner;
public class DateQuestion
{
	// 1 ,3 ,5 ,7 ,8 10 ,12
	public static ArrayList<data>ar;
	public static int []day1 = {31,28,31,30,31,30,31,31,30,31,30,31};
	public static int []day2 = {31,29,31,30,31,30,31,31,30,31,30,31};

	public static int[] num = new int[3];
	
	
	public static void myget(int a,int b,int c){
		if(a >= 60){
			a += 1900;
		}else{
			a += 2000;
		}
		if(b >12 || b <=0) return;
		int days;
		if(a % 4 == 0 && a % 100 != 0 || a % 400 == 0){
			days = day2[b - 1];
		}else{
			days = day1[b - 1];
		}
		if(c > days || c <= 0) return;
		
		data tmp = new data();
		tmp.year = a;
		tmp.month = b;
		tmp.day = c;
		ar.add(tmp);
		
		
		
	}
	public static void main(String[] args){
		ar = new ArrayList<data>();
		Scanner in = new Scanner(System.in);
		String tmp;
		tmp = in.next();
		String [] arr = new String[3];
		arr = tmp.split("/");
		
		for(int i=0;i<3;i++){
			num[i] = Integer.parseInt(arr[i]);
		}
		// 0 1 2
		
		myget(num[0],num[1],num[2]);
		myget(num[2],num[1],num[0]);
		myget(num[2],num[0],num[1]);

		
		Collections.sort(ar);
		
		boolean cf = true;
		while(cf){
			int sz = ar.size();
			cf = false;
			for(int i=0;i<sz;i++){
				for(int j=i+1;j<sz;j++){
					if(ar.get(i).compareTo(ar.get(j)) == 0 ){
						ar.remove(i);
						cf = true;
						j = sz; i = sz;
						break;
					}
				}
			}
		}
		for(int i=0;i<ar.size();i++){
			System.out.println(ar.get(i));
		}
	}
}

class data implements Comparable<data>{
	public int year;
	public int month;
	public int day;
	@Override
	public int compareTo(data o) {
		if(year == o.year){
			if(month == o.month){
				return day -o.day ;
			}
			return month - o.month;
		}
		return year-o.year;
	}
	public String toString(){
		String syear = String.valueOf(year);
		String smon = String.valueOf(month);
		if(smon.length() == 1){
			smon = "0" + smon;
		}
		String sday = String.valueOf(day);
		if(sday.length() == 1){
			sday = "0" + sday;
		}
		return syear + "-" + smon + "-" +sday;
	}
}

承压计算

水题，注意边界判断即可：

import java.util.Scanner;
import java.lang.Math;

class Cheng{
  public static double[][] arr = new double [100][100];
  public static int n = 29;
  
  public static void main(String[] args){
    Scanner sc = new Scanner(System.in);
    for(int i = 0 ; i < n ; i++)
      for(int j = 0 ; j <= i ; j++ ){
        arr[i][j] = sc.nextDouble();
      }
    for(int i = 1 ; i <= n ; i++)
      for(int j = 0 ; j <= i ; j++){
        if(j == 0) arr[i][j] += (arr[i-1][j]/2);
        else{
          arr[i][j] += arr[i-1][j-1]/2;
          arr[i][j] += arr[i-1][j]/2;
        }
      }
    double maxx = -1;
    double minn = 233333333;
    for(int i = 0 ; i <= n ; i++){
        maxx = Math.max(arr[29][i], maxx);
        minn = Math.min(arr[29][i], minn);
      }
    maxx *= ((double)2086458231/minn);
    System.out.println(maxx);
	}
}

k倍区间

首先统计前缀和sum[i] 表示A1+A2+…+Ai.所以对于任意一段区间[l,r]的和就是sum[r]-sum[l-1].如果要保证这个区间和为K倍数就是：(sum[r]-sum[l-1])%k == 0.变形后就是：sum[r]%k==sum[l-1]%k，所以我们计算前缀和的时候顺带模K，然后统计前缀和中相同的数据就行了。

import java.util.Scanner;

class Main{
  public static int[] arr = new int[100010];
  public static int[] bk = new int[100010];
  public static void main(String[] args){
    Scanner sc = new Scanner(System.in);
    int n,k;
    n = sc.nextInt();
    k = sc.nextInt();
    for(int i = 0 ; i < n ; i++)
      arr[i] = sc.nextInt();
    int sum = 0;
    arr[0] %= k;
    for(int i = 1 ; i < n ; i++)
      arr[i] = (arr[i-1]+arr[i])%k;
    for(int i = 0 ; i < n ; i++)
      sum += (bk[arr[i]]++);
	//如果前缀和有余数为0的需要加上
    System.out.println(sum+bk[0]);
  }
}

分巧克力

利用二分来判断来快速的判断巧克力的边长是否合适：

import java.util.Scanner;

public class Tang {

	public static int n,k;
	public static int []l;
	public static int []r;
	public static boolean test(int mid){
		int ans = 0;
		for(int i=0;i<n;i++){
			ans += (l[i] / mid) * (r[i]/mid);
		}
		if(ans >= k) return true;
		return false;
	}
	public static void main(String[] args) {
		Scanner in = new Scanner(System.in);
		n = in.nextInt();
		k = in.nextInt();
		 l= new int[n];
		 r= new int[n];
		int minn = 111111;
		for(int i=0;i<n;i++){
			l[i] = in.nextInt();
			r[i] = in.nextInt();
			minn = Math.min(minn,Math.min(l[i],r[i]));
		}
		int le = 1,ri = minn;
		int maxx = 1;
		int cc = 50;
		while(cc-- > 0){
			int mid = (le + ri)/2;
			if(test(mid)){
				maxx = Math.max(maxx,mid);
				le = mid;
			}else{
				ri = mid;
			}
		}
		System.out.println(maxx);	
	}
}
