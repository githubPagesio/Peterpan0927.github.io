Color a Tree

这是一道贪心的题目，意思就是说有一颗树，每个节点都有一个权重，涂完每个节点的消费等于涂完它后的当前时间乘节点的权重，每个节点都只需要涂一分钟，但是涂这个节点的时候必须要先涂它的父节点，起始时间为0，问要使涂完的消费最少，这个消费应该是多少？

最主要的就是看出他是一个贪心了，然后我们通过每次寻找最大权重点，然后将这个点和他的父节点合并，然后将权重也累加，每次循环判断最大权重点，直到把所有的节点都合并到了根节点，这样就可以保证我们的每一步都是最优的：

#include<stdio.h>
//节点的结构体
struct node  
{  
    int c,t,pre;
  	//节点的权重是c/t,一开始t都是1,读入的时候初始化为c
    double w;  
}s[1005];  

//找到非根节点的最大权重点
int find(int n,int r)  
{  
    double maxi=-1;  
    int p;  
    for(int i=1;i<=n;i++)  
    {  
        if(s[i].w > maxi && i != r)  
        {  
            p=i;  
            maxi=s[i].w;  
        }  
    }  
    return p;  
}  
int main()  
{  
    int n,r,i,a,b,ans,j,p,m;  
    while(scanf("%d%d",&n,&r),n|r)  
    {  
        ans=0;  
        for(i=1;i<=n;i++)  
        {  
            scanf("%d",&s[i].c);  
            s[i].t=1;  
            s[i].w=s[i].c;
            //因为根节点需要最先涂，所以在读入的时候就将那一分钟的消耗加上
            ans+=s[i].c;  
        }  
        for(i=1;i<n;i++)  
        {  
            scanf("%d%d",&a,&b);  
            s[b].pre=a;  
        }  
        for(i=1;i<n;i++)  
        {  
            m=find(n,r);  
            p=s[m].pre;  
            s[m].w=0;  
            for(j=1;j<=n;j++)  
                if(s[j].pre==m)
                  	//当前最大权重点和其父节点合并之后将子节点的父节点置为最大权重点的父节点
                    s[j].pre=p;  
            ans+=s[m].c*s[p].t;  
            s[p].c+=s[m].c;  
            s[p].t+=s[m].t; 
          	//重新计算权重
            s[p].w=1.0*s[p].c/s[p].t;  
        }  
        printf("%d\n",ans);  
    }  
    return 0;  
}

状压之铺砖块

有一个N*M(N<=5,M<=1000)的棋盘，现在有1*2及2*1的小木块无数个，要盖满整个棋盘，有多少种方式？答案只需要mod1,000,000,007即可。

例如：对于一个2*2的棋盘，有两种方法，一种是使用2个1*2的，一种是使用2个2*1的。

对于状态压缩的题目，关键之处就在于找到他的状态转移方程，这一道题的状态转移方程是：

for (int i=1;i<=M;i++)  
    {  
        for (int j=0;j<(1<<N);j++)  
        if (dp[i][j])  
        {  
            dfs(i,0,j,0);  
        }  
    }

所以最后的代码就是：

#include <cstdio>  
#include <cstring>  
#include <algorithm>  
#include <iostream>  
  
using namespace std;  
  
int N, M;  
long long dp[1005][34];  
  
void dfs(int i,int j,int state,int nex)  
{  
    if (j==N)  
    {  
  cout << nex << endl;
        dp[i+1][nex]+=dp[i][state];  
        return;  
    }  
    //如果这个位置已经被上一列所占用,直接跳过  
    if (((1<<j)&state)>0)  
        dfs(i,j+1,state,nex);  
    //如果这个位置是空的，尝试放一个1*2的  
    if (((1<<j)&state)==0)  
        dfs(i,j+1,state,nex|(1<<j));  
    //如果这个位置以及下一个位置都是空的，尝试放一个2*1的  
    if (j+1<N && ((1<<j)&state)==0 && ((1<<(j+1))&state)==0)  
        dfs(i,j+2,state,nex);  
    return;  
}  
  
int main()  
{  
    while (cin>>N>>M)  
    {  
        memset(dp,0,sizeof(dp));  
        if (N==0 && M==0) break;  
        dp[1][0]=1;   
        for (int i=1;i<=M;i++)  
        {  
            for (int j=0;j<(1<<N);j++)  
            if (dp[i][j])  
            {  
                dfs(i,0,j,0);  
            }  
        }
        for(int i = 1 ; i <= M+1 ;i++ ){
          for(int j = 0 ; j < N ; j++)
            cout << dp[i][j] << " ";
         cout << "\n";
        } 
        cout<<dp[M+1][0]<<endl;  
    }   
}

状态压缩之牛吃草

给出一个n行m列的草地，1表示肥沃，0表示贫瘠，现在要把任意数量牛放在肥沃的草地上，但是要求所有牛不能相邻，问有多少种放法。

搬运工写注释

#include<iostream>
#include<algorithm>
#include<stdio.h>
using namespace std;
int ans,n,m,dp[13][1<<13],map[13],st[1<<13];
int M=1e8;
int judge1(int x)//判断此状态是否有相邻的牛
{
    return x&(x<<1);
}
int  judge2(int i,int x)//判断此状态与地图是否冲突
{
    return map[i]&x;
}
int main()
{
    scanf("%d%d",&n,&m);
    for (int i=1;i<=n;i++)
    for (int j=1;j<=m;j++)
    {
        int x;
        scanf("%d",&x);
      	//状态转换，在判断的时候，1表示草地贫瘠
        if (x==0) map[i]+=(1<<(j-1));
    }
    int tot=0;
    //tot最后为最大的状态数
    for (int i=0;i<=(1<<m)-1;i++)
    if (!judge1(i))
    {
        tot++;
        st[tot]=i;
    }
    //对第一行进行预处理，得到可能情况存入dp数组
    for (int i=1;i<=tot;i++)
    {
        if (judge2(1,st[i])==0)
        dp[1][i]=1;
    }
    for (int i=2;i<=n;i++)
    {
        for (int j=1;j<=tot;j++)
        {
            if (judge2(i,st[j])) continue;
          	//遍历可能的状态，看看和上一行是否冲突
            for (int k=1;k<=tot;k++)
            {
                if (judge2(i-1,st[k])) continue;
                if ((st[j]&st[k])==0)
                {
                    dp[i][j] += dp[i-1][k];
                    dp[i][j] %= M;
                }
            }       
        }
    }
    for (int i=1;i<=tot;i++)
    {
        ans+=dp[n][i];
        ans%=M;
    }
    printf("%d\n",ans);
}

状态压缩之炮兵阵地

司令部的将军们打算在N*M的网格地图上部署他们的炮兵部队。一个N*M的地图由N行M列组成，地图的每一格可能是山地（用"H" 表示），也可能是平原（用"P"表示），如下图。在每一格平原地形上最多可以布置一支炮兵部队（山地上不能够部署炮兵部队）；一支炮兵部队在地图上的攻击范围如图中黑色区域所示：

#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;


int n,m,sta[105];
int map[105];
long long dp[105][65][65];
char s[15];

bool judge(int x){
  if((x&(x<<1))!=0 || (x&(x<<2))!=0) return 0;
  return 1;
}

int sum(int x){
  int su = 0;
  x = sta[x];
  for(int i = 0 ; i < m ; i++)
    if(((1<<i)&x)!=0) su++;
  return su;
}

int main(){
  scanf("%d %d", &n, &m);
  for (int i=1;i<=n;i++)
    {
        scanf("%s",s);
        for (int j=0;j<m;j++)
        {
            if (s[j]=='H')
            map[i]+=(1<<j);
        }
    }
  int tot = 0;
  for(int i = 0 ; i < (1 << m) ; i++){
    if(judge(i)){
      tot++;
      sta[tot] = i;
    }
  }
  memset(dp, 0 , sizeof(dp));
  //初始化dp数组第一行
  for(int i = 1 ; i <= tot ; i++)
    if((map[1]&sta[i])==0)
      dp[1][i][1] = sum(i);
  
  for(int i = 2 ; i <= n ; i++)
    //i-2行的状态
    for(int j = 1 ; j <= tot ; j++){
      if((map[i-2]&sta[j])!=0) continue;
      //i-1行的状态
      for(int k = 1 ; k <= tot ; k++){
        if((map[i-1]&sta[k])!=0) continue;
        if((sta[k]&sta[j])!=0) continue;
        //i行的状态
        for(int w = 1 ; w<= tot ; w++){
          if((map[i]&sta[w])!=0) continue;
          if((sta[w]&sta[k])!=0) continue;
          if((sta[w]&sta[j])!=0) continue;
          dp[i][w][k] = max(dp[i][w][k], dp[i-1][k][j]+sum(w));
        }
      }
    }
  long long ans = 0;
  for( int i = 1 ; i <= tot ;i++ )
    for(int j = 1 ; j<= tot ;j++ ){
      ans = max(dp[n][j][i],ans);
    }
    printf("%lld",ans);
  return 0;
}

手写迪杰斯特拉小试

#include<iostream>
#include<cstdio>
using namespace std;

const int maxnum = 100;
const int maxint = 999999;

void Dijkstra(int n, int v, int *dist, int*prev, int c[maxnum][maxnum]){
  bool s[maxnum];
  for(int i = 1 ; i <=n ; i++){
    dist[i] = c[v][i];
    s[i] = 0;
    if(dist[i] == maxint)
      prev[i] = 0;
    else
      prev[i] = v;
  }
  dist[v] = 0;
  s[v] = 1;
  for(int i = 2 ; i <= n ; i++){
    int tmp = maxint;
    int u = v;
    for(int j = 1 ; j <= n ; j++){
      //找没有放入集合离起点最近的点
      if(!s[j] && dist[j] < tmp){
        u = j;
        tmp = dist[j];
      }
    }
    s[u] = 1;
    for(int j = 1 ; j <= n ; j++){
      if(!s[j] && c[u][j] < maxint){
        int newdist = dist[u] + c[u][j];
        if(newdist < dist[j]){
          dist[j] = newdist;
          prev[j] = u;
        }
      }
    }
  }
}

int main(){
    // 各数组都从下标1开始
    int dist[maxnum];     // 表示当前点到源点的最短路径长度
    int prev[maxnum];     // 记录当前点的前一个结点
    int c[maxnum][maxnum];   // 记录图的两点间路径长度
    int n, line;             // 图的结点数和路径数
 
    // 输入结点数
    cin >> n;
    // 输入路径数
    cin >> line;
    int p, q, len;          // 输入p, q两点及其路径长度
 
    // 初始化c[][]为maxint
    for(int i=1; i<=n; ++i)
        for(int j=1; j<=n; ++j)
            c[i][j] = maxint;
 
    for(int i=1; i<=line; ++i)  
    {
        cin >> p >> q >> len;
        if(len < c[p][q])       // 有重边
        {
            c[p][q] = len;      // p指向q
            c[q][p] = len;      // q指向p，这样表示无向图
        }
    }
 
    for(int i=1; i<=n; ++i)
        dist[i] = maxint;
    for(int i=1; i<=n; ++i)
    {
        for(int j=1; j<=n; ++j)
            printf("%8d", c[i][j]);
        printf("\n");
    }
 
    Dijkstra(n, 1, dist, prev, c);
 
    // 最短路径长度
    cout << "源点到最后一个顶点的最短路径长度: " << dist[n] << endl;
  return 0;
}

一个BFS解决的事情

#include <iostream>
#include <queue>
using namespace std;
 
int N, K;
int line[100005];
 
void BFS()
{
    queue<int> que;
    que.push(N);
    line[N] = 0;
 
    while(!que.empty())
    {
        int tmp = que.front();
        que.pop();
 
        if(tmp == K)
            break;
 
        int next = tmp-1;
        if(next>=0 && !line[next])
        {
            que.push(next);
            line[next] = line[tmp] + 1;
        }
 
        next = tmp+1;
        if(next>=0 && !line[next])
        {
            que.push(next);
            line[next] = line[tmp] + 1;
        }
 
        next = tmp*2;
        if(next<=100000 && !line[next] && next-K<K-tmp)
        {
            que.push(next);
            line[next] = line[tmp] + 1;
        }
    }
}
 
int main()
{
    //freopen("input.txt", "r", stdin);
    while(scanf("%d %d", &N, &K) != EOF)
    {
        memset(line, 0, sizeof(line));
		BFS();
        printf("%d\n", line[K]);
 
    }
 
}
