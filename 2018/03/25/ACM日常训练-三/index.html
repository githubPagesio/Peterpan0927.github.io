Color a Tree

这是一道贪心的题目，意思就是说有一颗树，每个节点都有一个权重，涂完每个节点的消费等于涂完它后的当前时间乘节点的权重，每个节点都只需要涂一分钟，但是涂这个节点的时候必须要先涂它的父节点，起始时间为0，问要使涂完的消费最少，这个消费应该是多少？

最主要的就是看出他是一个贪心了，然后我们通过每次寻找最大权重点，然后将这个点和他的父节点合并，然后将权重也累加，每次循环判断最大权重点，直到把所有的节点都合并到了根节点，这样就可以保证我们的每一步都是最优的：

#include<stdio.h>
//节点的结构体
struct node  
{  
    int c,t,pre;
      //节点的权重是c/t,一开始t都是1,读入的时候初始化为c
    double w;  
}s[1005];  

//找到非根节点的最大权重点
int find(int n,int r)  
{  
    double maxi=-1;  
    int p;  
    for(int i=1;i<=n;i++)  
    {  
        if(s[i].w > maxi && i != r)  
        {  
            p=i;  
            maxi=s[i].w;  
        }  
    }  
    return p;  
}  
int main()  
{  
    int n,r,i,a,b,ans,j,p,m;  
    while(scanf("%d%d",&n,&r),n|r)  
    {  
        ans=0;  
        for(i=1;i<=n;i++)  
        {  
            scanf("%d",&s[i].c);  
            s[i].t=1;  
            s[i].w=s[i].c;
            //因为根节点需要最先涂，所以在读入的时候就将那一分钟的消耗加上
            ans+=s[i].c;  
        }  
        for(i=1;i<n;i++)  
        {  
            scanf("%d%d",&a,&b);  
            s[b].pre=a;  
        }  
        for(i=1;i<n;i++)  
        {  
            m=find(n,r);  
            p=s[m].pre;  
            s[m].w=0;  
            for(j=1;j<=n;j++)  
                if(s[j].pre==m)
                      //当前最大权重点和其父节点合并之后将子节点的父节点置为最大权重点的父节点
                    s[j].pre=p;  
            ans+=s[m].c*s[p].t;  
            s[p].c+=s[m].c;  
            s[p].t+=s[m].t; 
              //重新计算权重
            s[p].w=1.0*s[p].c/s[p].t;  
        }  
        printf("%d\n",ans);  
    }  
    return 0;  
}

手写迪杰斯特拉小试

#include<iostream>
#include<cstdio>
using namespace std;

const int maxnum = 100;
const int maxint = 999999;

void Dijkstra(int n, int v, int *dist, int*prev, int c[maxnum][maxnum]){
  bool s[maxnum];
  for(int i = 1 ; i <=n ; i++){
    dist[i] = c[v][i];
    s[i] = 0;
    if(dist[i] == maxint)
      prev[i] = 0;
    else
      prev[i] = v;
  }
  dist[v] = 0;
  s[v] = 1;
  for(int i = 2 ; i <= n ; i++){
    int tmp = maxint;
    int u = v;
    for(int j = 1 ; j <= n ; j++){
      //找没有放入集合离起点最近的点
      if(!s[j] && dist[j] < tmp){
        u = j;
        tmp = dist[j];
      }
    }
    s[u] = 1;
    for(int j = 1 ; j <= n ; j++){
      if(!s[j] && c[u][j] < maxint){
        int newdist = dist[u] + c[u][j];
        if(newdist < dist[j]){
          dist[j] = newdist;
          prev[j] = u;
        }
      }
    }
  }
}

int main(){
    // 各数组都从下标1开始
    int dist[maxnum];     // 表示当前点到源点的最短路径长度
    int prev[maxnum];     // 记录当前点的前一个结点
    int c[maxnum][maxnum];   // 记录图的两点间路径长度
    int n, line;             // 图的结点数和路径数

    // 输入结点数
    cin >> n;
    // 输入路径数
    cin >> line;
    int p, q, len;          // 输入p, q两点及其路径长度

    // 初始化c[][]为maxint
    for(int i=1; i<=n; ++i)
        for(int j=1; j<=n; ++j)
            c[i][j] = maxint;

    for(int i=1; i<=line; ++i)  
    {
        cin >> p >> q >> len;
        if(len < c[p][q])       // 有重边
        {
            c[p][q] = len;      // p指向q
            c[q][p] = len;      // q指向p，这样表示无向图
        }
    }

    for(int i=1; i<=n; ++i)
        dist[i] = maxint;
    for(int i=1; i<=n; ++i)
    {
        for(int j=1; j<=n; ++j)
            printf("%8d", c[i][j]);
        printf("\n");
    }

    Dijkstra(n, 1, dist, prev, c);

    // 最短路径长度
    cout << "源点到最后一个顶点的最短路径长度: " << dist[n] << endl;
  return 0;
}

一个BFS解决的事情

#include <iostream>
#include <queue>
using namespace std;

int N, K;
int line[100005];

void BFS()
{
    queue<int> que;
    que.push(N);
    line[N] = 0;

    while(!que.empty())
    {
        int tmp = que.front();
        que.pop();

        if(tmp == K)
            break;

        int next = tmp-1;
        if(next>=0 && !line[next])
        {
            que.push(next);
            line[next] = line[tmp] + 1;
        }

        next = tmp+1;
        if(next>=0 && !line[next])
        {
            que.push(next);
            line[next] = line[tmp] + 1;
        }

        next = tmp*2;
        if(next<=100000 && !line[next] && next-K<K-tmp)
        {
            que.push(next);
            line[next] = line[tmp] + 1;
        }
    }
}

int main()
{
    while(scanf("%d %d", &N, &K) != EOF)
    {
        memset(line, 0, sizeof(line));
        BFS();
        printf("%d\n", line[K]);
    }

}
