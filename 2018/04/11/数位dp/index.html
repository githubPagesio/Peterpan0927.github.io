数位dp

dp模版

通常数位dp和记忆化dfs是不分家的，如果dfs还没有写会还是先练练dfs吧

思想介绍

其实所谓的数位dp就是基于数字的每一位的状态来转移的dp，经历过之前的题目的同学说到这里其实就比较好理解了，通常的数位dp可以写成记忆化搜索的模版的形式：

int dfs(int i, int s, bool e){
  if(i==-1) return s==target_s;
  if(!e && f[i][s]!=-1) return f[i][s];
  int res = 0;
  int u = e ? num[i] : 9;
  for(int d = first?1:0 ; d<=u ; d++)
    res += dfs(i-1, new_s(s,d),e&&d==u);
  return e?res:f[i][s]=res;
}

其中的f是记忆化数组，初始化为-1，i 为当前处理串的第i位，表示还有i+1位等待处理，s是之前数字的状态，如果有其他限制条件就继续往上加，e表示之前的数是否是上界的前缀(也就是后面的数是不是可以随便填)。因为这就是个模版，所以直接上题来体验一下就好了。

HDU_3555

#include<iostream>
#include<cstring>
#include<algorithm>
#include<cstdio>
using namespace std;

typedef long long LL;
LL n;
LL f[31][2][2];
int digit[35];

LL dfs(int len,int have,int s,int e){
  if(len == -1) return have;
  if(!e && f[len][have][s] != -1) return f[len][have][s];

  LL ans = 0;
  int u = e ? digit[len] : 9;
  for(int d = 0 ; d <= u ;d++){
    if(s && d == 9) ans += dfs(len-1, 1, d==4, e && d == u);
    else ans += dfs(len-1,have,d==4, e && d == u);
  } 
  return e ? ans : f[len][have][s] = ans;
}

LL solve(LL x){
  memset(digit, 0, sizeof(digit));
  int i = 0;
  while(x){
    digit[i++]=x%10;
    x /= 10;
  }
  return dfs(i-1,0,0,1);
}

int main(){
  int t;
  cin >> t;
  while(t--){
    cin >> n;
    memset(f,-1,sizeof(f));
    cout << solve(n) << endl;
  }
  return 0;
}

HDU_2089

#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;

int n,m;
int f[8][2];
int digit[10];

int cal_len(int x){
  int sum = 0;
  while(x){
    sum += 1;
    x /= 10;
  }
  return sum;
}

void cal_digit(int x, int len){
  memset(digit,0,sizeof(digit));
  for(int i = 1 ; i <= len ;i++){
    digit[i] = x%10;
    x /= 10; 
  }
}

int dfs(int len, int s, int e){
  if(len == 0) return 1;
  if(!e && f[len][s]!=-1) return f[len][s];

  int ans = 0;
  int u = e ? digit[len] : 9;
  //因为车子牌号不是数字，首位可以是0
  for(int i = 0 ;  i<= u ;i++){
    if(i == 4 || (s && i == 2)) continue;
    ans += dfs(len-1, i == 6, e && i == u);
  }
  return e ? ans : f[len][s] = ans;
}

int solve(int x){
  int len  = cal_len(x);
  cal_digit(x, len);

  int res = dfs(len,0,1);
  return res;
}


int main(){
  memset(f,-1,sizeof(f));
  while(cin >> n >> m && n+m){
    cout << solve(m) - solve(n-1) << endl;
  }
  return 0;
}

HDU_3652

因为有两个判断条件，所以加一维

#include<iostream>
#include<cstring>
#include<algorithm>

#define Memset(a,b) memset(a,b,sizeof(a));

using namespace std;

int digit[16];
int f[16][16][3];


int dfs(int len,int mod, int s ,int e){
    if(len == 0) return s == 2 && mod == 0;
    if(!e && f[len][mod][s] != -1) return f[len][mod][s];
    int u = e ? digit[len] : 9;
    int ans = 0;
    for(int d = 0; d <= u ; d++){
        int _mod = (mod*10 + d)%13;
        int _s = s;
        if(s == 0 && d == 1) _s = 1;
        if(s == 1 && d != 1) _s = 0;
        if(s == 1 &&  d == 3) _s = 2;
        ans += dfs(len-1, _mod, _s ,e && u==d);
    }
    return e ? ans : f[len][mod][s] = ans;
}

int solve(int x){
    int len = 0;
    memset(digit,0,sizeof(digit));
    while(x){
      	len += 1;
        digit[len] = x%10;
        x /= 10;
    }
    dfs(len,0,0,1);
}


int main(){
    int t;
    while(cin >>t){
        memset(f,-1,sizeof(f));
        cout << solve(t) << endl; 
    }
}

那么经过这个之后我们是不是就可以想到对于第一题在于空间上进行一次优化，因为判断条件是可以叠加的，所有我们只需要修改关键地方的代码即可：

LL dfs(int len,int s,int e){
  if(len == 0) return s == 2;
  if(!e && f[len][s] != -1) return f[len][s];

  LL ans = 0;
  int u = e ? digit[len] : 9;
  for(int d = 0 ; d <= u ;d++){
    int _s = s;
    if(s == 0 && d == 4) _s = 1;
    if(s == 1 && d != 4) _s = 0;
    if(s == 1 && d == 9) _s = 2;
    ans += dfs(len-1, _s , e && u==d);
  }
  return e ? ans : f[len][s] = ans;
}

HDU_3709

因为支点不好在数位dp里面进行枚举，所以我们在外层通过一个循环来枚举，dfs中固定支点

#include<iostream>
#include<algorithm>
#include<cstring>
#include<cstdio>
using namespace std;

#define MAXN 20
typedef long long LL;
int t;
LL n,m;
int digit[MAXN];
LL f[MAXN][MAXN][2005];

LL dfs(int len, int o, int s,int e){
  if(len <= 0) return s == 0;
  if(s < 0) return 0;
  if(!e && f[len][o][s] != -1) return f[len][o][s];
  int u = e ? digit[len] : 9;
  LL ans = 0;
  for(int d = 0 ; d <= u ; d++){
    int next = s+(len-o)*d;
    ans += dfs(len-1,o,next,e&&(u==d));
  }
  if(!e) f[len][o][s] = ans;
  return ans;
}

LL solve(LL x){
  memset(digit,0,sizeof(digit));
  int len = 0;
  while(x){
    len += 1;
    digit[len] = x%10;
    x /= 10;
  }
  LL ans = 0;
  for(int i = 1 ; i <= len ; i++)
    ans += dfs(len,i,0,1);
  return ans - (len-1);
}

int main(){
  scanf("%d", &t);
  while(t--){
    memset(f,-1,sizeof(f));
    scanf("%lld%lld", &n, &m);
    LL ans = solve(m) - solve(n-1);
    printf("%lld\n", ans);
  }
  return 0;
}

POJ_3252

主要的难度在于前导0的处理上，对于回归条件的判断。

#include<iostream>
#include<cstring>
#include<algorithm>

typedef long long LL;

using namespace std;

int digit[50];
LL f[50][50][50];


LL dfs(int len,int zero, int one ,int e,int fi){
    if(len == 0){
        if(fi) return 1;
        if(zero >= one) return 1;
        return 0;
    }
    if(!e && !fi && f[len][zero][one] != -1) return f[len][zero][one];
    int u = e ? digit[len] : 1;
    LL ans = 0;
    for(int d = 0; d <= u ; d++){
        if(fi){
            if(d==0) ans+=dfs(len-1,0,0,e&&(u==d),1);
            else ans+=dfs(len-1,zero,one+1,e&&(u==d),0);
        }else{
            if(d==0)  
                ans+=dfs(len-1,zero+1,one,e&&u==d,0);//在二进制后面放0  
            else  
                ans+=dfs(len-1,zero,one+1,e&&u==d,0);//放1  
        }
    }
    return (e&&fi) ? ans : f[len][zero][one] = ans;
}

LL solve(LL x){
    int len = 0;
    memset(digit,0,sizeof(digit));
    while(x){
        digit[++len] = x & 1;
        x >>= 1;
    }
    return dfs(len,0,0,1,1);
}


int main(){
    LL n,m;
    while(cin >> n >> m){
        memset(f,-1,sizeof(f));
        cout << solve(m)-solve(n-1) << endl; 
    }
    return 0;
}
