Bone Collector

一维数组，空间上优化

#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;

int main(){
  int t;
  cin >> t;
  while(t--){
    int n,V,v[1005],val[1005],dp[1005];
    cin >> n >> V;
    for(int i=0; i<n; i++)
        scanf("%d",&val[i]);
    for(int i=0; i<n; i++)
        scanf("%d",&v[i]);
    memset(dp, 0 , sizeof(dp));
    for(int i = 0 ; i < n ; i++)
      for(int j = V ; j >= v[i] ; j--)
        dp[j] = max(dp[j], dp[j-v[i]]+val[i]);
    cout << dp[V] << endl;    
  }
  return 0;
}

Piggy Bank

#include<iostream>
#include<cstdio>
#include<algorithm>

#define loop(x,t,n) for (int x = t; x < n; ++x)  

int d[10005],c[505],v[505];

using namespace std;

int main(){
  int t;
  cin >> t;
  while(t--){
    int E,F,n;
    cin >> E >> F;
    int V = F-E;
    cin >> n;
    loop(i,0,n)
      cin >> v[i] >> c[i];
    loop(i,1,V+1)
      d[i]=5000000;
    d[0] = 0;
    loop(i,0,n)
      loop(j,c[i],V+1)
        d[j] = min(d[j-c[i]] + v[i], d[j]);
    if(d[V] == 5000000)
      cout << "This is impossible." << endl;
    else cout << "The minimum amount of money in the piggy-bank is " << d[V] <<"."<< endl;
  }
  return 0;
}

Big Event in HDU

多重背包，母函数

#include<stdio.h>
#include<string.h>

int bag[333333];

int main()
{
	int n,w[555];
	while(scanf("%d",&n)&&(n>=0))
	{
		int i,j,sum=0,size,k=1;;
		for(i=1;i<=n;i++)
		{
			int tmp,v,m;
			scanf("%d %d",&v,&m);
			sum+=v*m;
			tmp=1;
			while(m>tmp)
			{
				w[k++]=tmp*v;
				m-=tmp;
				tmp<<=1;
			}
			w[k++]=m*v;
		}
		size=sum/2;
		memset(bag,0,sizeof(bag));
		for(i=1;i<k;i++)
			for(j=size;j>=w[i];j--)
				if(bag[j]<bag[j-w[i]]+w[i])
					bag[j]=bag[j-w[i]]+w[i];
		printf("%d %d\n",sum-bag[size],bag[size]);
	}
	return 0;
}
