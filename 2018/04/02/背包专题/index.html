背包是dp下的一个小分支，普遍层次的背包包括01背包，完全背包，多重背包以及三种混合的背包，已经可以涵盖绝大多数的问题抽象，在这之上又延伸了一些背包问题的变种，进一步泛化了物品的概念，如背包分组等，下面通过HDU上的例题来了解一下这个小专题。(不管在什么时候一定要注意dp数组的初始化以及是否需要预处理)

Dividing

一个很好的动归模版，很多的背包都可以直接拿这个模版来用。

#include<iostream>

#include<cstring>

int a[7];
int sum;
int dp[120005];

void ZeroOnePack(int cost,int weight){
  for(int j = sum ; j >= cost ; j--)
    dp[j] = max(dp[j], dp[j-cost]+weight);
}

void CompletePack(int cost, int weight){
  for(int j = cost ; j <= sum ; j++)
    dp[j] = max(dp[j], dp[j-cost]+weight);
}

void MutiplyPack(int cost, int weight, int amount){
  if(cost*amount >= sum){
    CompletePack(cost, weight);
    return;
  }
  int tmp = 1;
  while(tmp < amount){
    ZeroOnePack(tmp*cost, tmp*weight);
    amount -= tmp;
    tmp <<= 1;
  }
  ZeroOnePack(amount*cost, amount*weight);
}

金明的预算方案

有依赖的背包转换为分组背包，因为每个主件底下可能会有附件，其实我们在选择一个主件和其附属的附件的时候，选择是互斥的，也就是说我们只会有一种选择，于是我们将每一种选择枚举出来，然后通过分组背包的思想就可以解决问题了

#include<cstdio>  
#include<algorithm>  
using namespace std;  

const int max_n=3200;  
const int max_m=60;  

int n,m,s[max_m];  
int v[max_m+20][4],w[max_m+20][4];  
int f[max_n+20];  

int main()  
{  
  int i,j,k,a,b,c;  
  scanf("%d%d",&n,&m),n/=10;  
  for(i=1;i<=m;i++)  
    {  
      scanf("%d%d%d",&a,&b,&c);  
      if(c==0)k=s[i],w[i][k]=a*b,v[i][k]=a/10,s[i]++;  
      else k=s[c],w[c][k]=a*b,v[c][k]=a/10,s[c]++;  
    }  

  //预处理
  for(i=1;i<=m;i++)  
    {  
      if(s[i]==3)  
        {  
          s[i]++;  
          v[i][3]=v[i][1]+v[i][2]+v[i][0];  
          w[i][3]=w[i][1]+w[i][2]+w[i][0];  
        }  
      if(s[i]>=2)v[i][1]+=v[i][0],w[i][1]+=w[i][0];  
      if(s[i]>=3)v[i][2]+=v[i][0],w[i][2]+=w[i][0];  
    }  
      //关键dp
    for(i=1;i<=m;i++) 
     //分组背包固定体积枚举组内物品
     for(k=n;k>=1;k--)  
       for(j=0;j<s[i];j++)if(k>=v[i][j])  
         f[k]=max(f[k],f[k-v[i][j]]+w[i][j]);  
  printf("%d\n",f[n]);  
  return 0;  
}

搭建双塔

这是一个01背包加上差值dp的题目，因为要求最大，我们通过一个f[i][j]数组来表示放第i块水晶的时候较高的那座塔的高度，这个时候我们从上一种状态转移过来有四种方式，也就是这一块水晶可能会造成四种不同的状态(不是很会讲算法)，那我们就直接来一个个说明吧：

  1. 这一个水晶不需要放：f[i-1][j]
  2. 放在较矮的那座塔，但是高度没有超过较高的那座塔：f[i-1][j+a[i]]
  3. 放在较高的那座塔上：f[i-1][j-a[i]]+a[i]
  4. 放在较矮的那座塔，高度超过较高的那座塔，更新高度：f[i-1][a[i]-j]+j

所以代码表示就可以得到了：

#include <cstdio>  
#include <cstring>  
#include <iostream>  
#include <algorithm>  
using namespace std;  
const int INF=0x3f3f3f3f;  
int a[110];//水晶高度  
int f[110][2010];//前i个水晶搭成的两座塔高度差为j时高塔的高度。  
int main()  
{  
    int n,i,j,sum=0;  
    scanf("%d",&n);  
    for(i=1;i<=n;++i)  
    {  
        scanf("%d",&a[i]);  
        sum+=a[i];  
    }  
    memset(f,-INF,sizeof(f));//初始时候，只有f[0][0]=0是合法的f[0][j]都不合法，要设为负无穷  
    f[0][0]=0;  
    for(i=1;i<=n;++i)  
    {  
        for(j=0;j<=sum;++j)  
        {  
            f[i][j]=max(f[i-1][j],f[i-1][j+a[i]]);//不放，或者放在矮的塔上，并且不超过高的塔  
            if(j>=a[i])  
                f[i][j]=max(f[i][j],f[i-1][j-a[i]]+a[i]);//放高的上
            else  
                f[i][j]=max(f[i][j],f[i-1][a[i]-j]+j);//放矮的上且超过了高的  
        }  

    }  
    if(f[n][0]>0) printf("%d",f[n][0]);  
    else printf("Impossible");  
    return 0;  
}

守望者的逃离

优化版本，因为有三种选择，所以不可能在一次循环中解决问题，所以我们先一直闪现，然后用跑步来优化其中一部分情况，因为距离很长，所以我们只能通过枚举时间。

#include <iostream>
#include <cstdio>
#include <cstring>
#define MAXT 300001
using namespace std;

int magic,dis,tim,ans;
int f[MAXT];  //f[i]表示当时间为i时的最优解 
bool flag;    //flag表示能否逃出荒岛 

void dp()  //核心代码，dp 
{
    for (int i=1;i<=tim;i++){  //这个循环先处理不停闪现的情况，原则是有魔法就闪，没有就停下来补 
        if (magic>=10){        //够魔法值，进行闪现 
            f[i]=f[i-1]+60;
            magic-=10;
        }else{      //不够魔法值，停下来补充 
            f[i]=f[i-1];
            magic+=4;
        }
    }

    for (int i=1;i<=tim;i++){  //对于所有时间，看是否能用普通跑步来优化 
        f[i]=max(f[i],f[i-1]+17);
    }
}

void find_ans()  //找到答案（不重要） 
{
    for (int i=1;i<=tim;i++){
        if (f[i]>=dis){
            flag=true;
            ans=i;
            break;
        }
        if (i==tim){
            flag=false;
            ans=f[i];
        }
    }

    if (flag){
        printf("Yes\n%d\n",ans);
    }else{
        printf("No\n%d\n",ans);
    }
}

int main()
{
    memset(f,0,sizeof(f));
    scanf("%d %d %d",&magic,&dis,&tim);
    dp();
    find_ans();
    return 0;
}

拼拼图的小衫

这个题目直接使用f[i,j]表示第i个集合还有j的剩余时候可以放入的最大拼图数显然不合理，因为还要满足不交叉，所以我们换一种思路，通过f[j]表示放入第j个数最少需要多少个集合，这个集合中已经放了多少空间：

f[j] = (t,s)

#include<cstdio>
const int maxn=1000;
int n,m,T,a[maxn+10];
struct node{
    int t,s;
    bool operator < (const node d) const{
        if (t<d.t)
            return 1;
        if (t>d.t)
            return 0;
        return s<d.s;
    }
    node operator + (const int d) const{
        if (s+d<=T)
            return (node){t,s+d};
        else
            return (node){t+1,d};
    }
};
node f[maxn+10];
int main(){
    scanf("%d%d%d",&n,&m,&T);
    node te;
    for (int i=1;i<=n;i++)
        scanf("%d",&a[i]);
    f[0].t=1;
    for (int i=1;i<=n;i++){
        f[i]=f[i-1]+a[i];
        for (int j=i-1;j>=1;j--){
            te=f[j-1]+a[i];
            if (te<f[j])
                f[j]=te;
        }
    }
    for (int i=n;i>=0;i--)
        if (f[i].t<=m){

        }

兴奋剂检查

一开始的时候以为是一个01背包，然后我通过枚举药品的个数还有检测的指标个数，最后再通过药品加成做一次01背包，但是这样只能过一部分数据，最后看答案发现是要将多重条件压缩然后做01背包。因为原来的那种写法相当把所有的限制条件叠到一个背包上了。：

#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
int n, m;
int p[6], f[6];
int dp[5001000];

//做一次hash
int h(int a, int b, int c, int d, int e){
  return a*(p[2]+1)*(p[3]+1)*(p[4]+1)*(p[5]+1) + b*(p[3]+1)*(p[4]+1)*(p[5]+1) + c*(p[4]+1)*(p[5]+1) + d*(p[5]+1) + e;
}

int main() {
    scanf("%d %d", &n, &m);
    memset(p, 0, sizeof(p));
    for (int i = 1; i <= m; i++) {
        scanf("%d", &p[i]);
    }
    int x;
    int ans = 0;
    for (int i = 1; i <= n; i++) {
        scanf("%d", &x);
        memset(f, 0, sizeof(f));
        for (int j = 1; j <= m; j++) scanf("%d", &f[j]);
        for (int a = p[1]; a >= f[1]; a--) {
            for (int b = p[2]; b >= f[2]; b--) {
                for (int c = p[3]; c >= f[3]; c--) {
                    for (int d = p[4]; d >= f[4]; d--) {
                        for (int e = p[5]; e >= f[5]; e--) {
                            int now = h(a, b, c, d, e), pre = h(a-f[1], b-f[2], c-f[3], d-f[4], e-f[5]);
                            dp[now] = max(dp[now], dp[pre] + x);
                            ans = max(ans, dp[now]);
                        }
                    }
                }
            }
        }
    }
    printf("%d\n", ans);
    return 0;
}

高性能计算机

算法导论上的一道题，用到二次dp，还有一些高级优化等待进一步看黑书，首先看看low逼的版本怎么做吧:

首先我们可以划分成两个子问题：

  1. 算出第i个计算节点完成ai个任务A和bi个任务B所需要的最短时间
  2. 算出前i个节点完成ai个任务和bi个任务所需要的最短时间，计算之前需要预处理一下。

前者很简单，后者的状态转移方程要首先对前i-1个节点的最短时间和第i个节点的最短时间取一个最大值然后比较，因为在操作系统中两个并行任务计算完成时间肯定是取最晚完成的那个。

#include <iostream>
#include <cstring>
#include <algorithm>
using namespace std;

int n,na,nb,ta[61],tb[61],ka[61],kb[61];
int tim[21][61][61],dp[21][61][61];

inline void calc(int x){
    int t[2][61][61];
    memset(t,0x3f,sizeof(t));
    //初始化
    t[1][0][0] = t[0][0][0] = 0;
    for(int i = 0 ; i<= na ;i++){
        for(int j = 0 ; j <= nb ;j++){
            for(int k = 1 ; k<=i;k++) t[0][i][j] = min(t[1][i-k][j]+ta[x]+ka[x]*k*k, t[0][i][j]);
            for(int k = 1 ; k<=j;k++) t[1][i][j] = min(t[1][i][j], t[0][i][j-k]+tb[x]+kb[x]*k*k);
            tim[x][i][j] = min(t[1][i][j], t[0][i][j]);
        }

    }
}

int main(){
    cin >> na >> nb;
    cin >> n;
    for(int i = 1 ;i <= n ;i++)
        cin >> ta[i] >> tb[i] >> ka[i] >> kb[i];
    for(int i = 1 ; i<= n ;i++) calc(i);

    memset(dp , 0x3f,sizeof(dp));

    //预处理
    for(int i = 0 ; i<=na ;i++)
        for(int j = 0 ;j <=nb ;j++)
            dp[1][i][j] = tim[1][i][j];

    for(int x = 2 ; x <= n ; x++)
        for(int i = 0 ;i <= na ;i++)
            for(int j = 0 ; j <= nb ;j++ )
                for(int ki = 0 ; ki <= i ; ki++)
                    for(int kj = 0 ; kj <= j ;kj++)
                        dp[x][i][j] = min(dp[x][i][j], max(dp[x-1][i-ki][j-kj], tim[x][ki][kj]));
    cout << dp[n][na][nb] << endl;
}
