Bone Collector

一维数组，空间上优化

#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;

int main(){
  int t;
  cin >> t;
  while(t--){
    int n,V,v[1005],val[1005],dp[1005];
    cin >> n >> V;
    for(int i=0; i<n; i++)
        scanf("%d",&val[i]);
    for(int i=0; i<n; i++)
        scanf("%d",&v[i]);
    memset(dp, 0 , sizeof(dp));
    for(int i = 0 ; i < n ; i++)
      for(int j = V ; j >= v[i] ; j--)
        dp[j] = max(dp[j], dp[j-v[i]]+val[i]);
    cout << dp[V] << endl;    
  }
  return 0;
}

Piggy Bank

#include<iostream>
#include<cstdio>
#include<algorithm>

#define loop(x,t,n) for (int x = t; x < n; ++x)  

int d[10005],c[505],v[505];

using namespace std;

int main(){
  int t;
  cin >> t;
  while(t--){
    int E,F,n;
    cin >> E >> F;
    int V = F-E;
    cin >> n;
    loop(i,0,n)
      cin >> v[i] >> c[i];
    loop(i,1,V+1)
      d[i]=5000000;
    d[0] = 0;
    loop(i,0,n)
      loop(j,c[i],V+1)
        d[j] = min(d[j-c[i]] + v[i], d[j]);
    if(d[V] == 5000000)
      cout << "This is impossible." << endl;
    else cout << "The minimum amount of money in the piggy-bank is " << d[V] <<"."<< endl;
  }
  return 0;
}

Big Event in HDU

多重背包，母函数

#include<stdio.h>
#include<string.h>

int bag[333333];

int main()
{
	int n,w[555];
	while(scanf("%d",&n)&&(n>=0))
	{
		int i,j,sum=0,size,k=1;;
		for(i=1;i<=n;i++)
		{
			int tmp,v,m;
			scanf("%d %d",&v,&m);
			sum+=v*m;
			tmp=1;
			while(m>tmp)
			{
				w[k++]=tmp*v;
				m-=tmp;
				tmp<<=1;
			}
			w[k++]=m*v;
		}
		size=sum/2;
		memset(bag,0,sizeof(bag));
		for(i=1;i<k;i++)
			for(j=size;j>=w[i];j--)
				if(bag[j]<bag[j-w[i]]+w[i])
					bag[j]=bag[j-w[i]]+w[i];
		printf("%d %d\n",sum-bag[size],bag[size]);
	}
	return 0;
}

Dividing

一个很好的动归模版，很多的背包都可以直接拿这个模版来用。

#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;


int a[7];
int sum;
int dp[120005];

void ZeroOnePack(int cost,int weight){
  for(int j = sum ; j >= cost ; j--)
    dp[j] = max(dp[j], dp[j-cost]+weight);
}

void CompletePack(int cost, int weight){
  for(int j = cost ; j <= sum ; j++)
    dp[j] = max(dp[j], dp[j-cost]+weight);
}

void MutiplyPack(int cost, int weight, int amount){
  if(cost*amount >= sum){
    CompletePack(cost, weight);
    return;
  }
  int tmp = 1;
  while(tmp < amount){
    ZeroOnePack(tmp*cost, tmp*weight);
    amount -= tmp;
    tmp <<= 1;
  }
  ZeroOnePack(amount*cost, amount*weight);
}

int main(){
  int i,p=1;
    while(1)
    {
        sum=0;
        for(i=1;i<=6;i++)
        {
            scanf("%d",&a[i]);
            sum+=a[i]*i;
        }
        if(sum==0) return 0;
        printf("Collection #%d:\n",p++);
        if(sum%2==1)
        {
            printf("Can't be divided.\n\n");
            continue;
        }
        sum /= 2;
        memset(dp,0,sizeof(dp));
        for(i=1;i<=6;i++)
          MutiplyPack(i,i,a[i]);
        if(dp[sum] == sum)
          printf("Can be divided.\n\n");
        else
          printf("Can't be divided.\n\n");
    }
  return 0;
}

FATE

二维完全背包

#include <iostream>  
#include <stdio.h>  
#include <string.h>  
#include <stdlib.h>  
  
using namespace std;  
  
int Max (int a,int b)  
{  
    return a > b ? a : b;  
}  
  
int main()  
{  
    int n,m,s,k;  
    int dp[100][100];  
    int val[100],wei[100];  
    while(scanf("%d%d%d%d",&n,&m,&k,&s)!=EOF)///经验值 忍耐度 种类 杀怪数  
    {  
        for(int i=0; i<k; i++)  
        {  
            scanf("%d%d",&val[i],&wei[i]);  
        }  
        memset(dp,0,sizeof(dp));  
        for(int i = 0; i < k; i++) ///种类  
        {  
            for(int j = wei[i]; j <= m; j++) ///忍耐度  
            {  
                for(int k = s; k >= 1; k--) ///杀怪数  
                {  
                    dp[j][k] = Max(dp[j][k],dp[j-wei[i]][k-1] + val[i]);///在忍耐度为j，杀了k个怪的状态下所对应的经验值  
                }  
            }  
        }  
        int i;  
        for (i =0 ;i <= m; i++)  
        {  
            if (dp[i][s] >= n)  
                break;  
        }  
        if (i > m)  
            printf ("-1\n");  
        else printf ("%d\n",m - i);  
    }  
    return 0;  
}
