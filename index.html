Color a Tree

这是一道贪心的题目，意思就是说有一颗树，每个节点都有一个权重，涂完每个节点的消费等于涂完它后的当前时间乘节点的权重，每个节点都只需要涂一分钟，但是涂这个节点的时候必须要先涂它的父节点，起始时间为0，问要使涂完的消费最少，这个消费应该是多少？

最主要的就是看出他是一个贪心了，然后我们通过每次寻找最大权重点，然后将这个点和他的父节点合并，然后将权重也累加，每次循环判断最大权重点，直到把所有的节点都合并到了根节点，这样就可以保证我们的每一步都是最优的：

#include<stdio.h>
//节点的结构体
struct node  
{  
    int c,t,pre;
  	//节点的权重是c/t,一开始t都是1,读入的时候初始化为c
    double w;  
}s[1005];  

//找到非根节点的最大权重点
int find(int n,int r)  
{  
    double maxi=-1;  
    int p;  
    for(int i=1;i<=n;i++)  
    {  
        if(s[i].w > maxi && i != r)  
        {  
            p=i;  
            maxi=s[i].w;  
        }  
    }  
    return p;  
}  
int main()  
{  
    int n,r,i,a,b,ans,j,p,m;  
    while(scanf("%d%d",&n,&r),n|r)  
    {  
        ans=0;  
        for(i=1;i<=n;i++)  
        {  
            scanf("%d",&s[i].c);  
            s[i].t=1;  
            s[i].w=s[i].c;
            //因为根节点需要最先涂，所以在读入的时候就将那一分钟的消耗加上
            ans+=s[i].c;  
        }  
        for(i=1;i<n;i++)  
        {  
            scanf("%d%d",&a,&b);  
            s[b].pre=a;  
        }  
        for(i=1;i<n;i++)  
        {  
            m=find(n,r);  
            p=s[m].pre;  
            s[m].w=0;  
            for(j=1;j<=n;j++)  
                if(s[j].pre==m)
                  	//当前最大权重点和其父节点合并之后将子节点的父节点置为最大权重点的父节点
                    s[j].pre=p;  
            ans+=s[m].c*s[p].t;  
            s[p].c+=s[m].c;  
            s[p].t+=s[m].t; 
          	//重新计算权重
            s[p].w=1.0*s[p].c/s[p].t;  
        }  
        printf("%d\n",ans);  
    }  
    return 0;  
}

Uva_1592 dataBase

考虑到字符串匹配的太慢，所以通过map和pair做一个二元组映射，然后枚举列，因为列最大只有10，然后将行从上到下扫描，看看有没有相同的即可：

#include <iostream>
#include <map>
#include <string>
#include <cstdio>

using namespace std;
const int maxn = 10000+5;
string s[maxx][15];
map <string, int>str_id;
tydedef pair<int, int>p;
int n,m;


int main(){
  while(cin>>n>>m)
    {
       str_id.clear();
        getchar();
        //read
        for(int i=1;i<=n;i++)
        {
            for(int j=1;j<=m;j++)
            {
                s[i][j]="";
                while(1)
                {
                    char c=getchar();
                    if(c==','||c=='\n')break;
                     s[i][j]+=c;
                }
                str_id[s[i][j]]=j*100000+i;//id
            }
        }
        //solve
        int flag=1;
        for(int i=1;i<m;i++)
        {
            for(int j=i+1;j<=m;j++)
            {
                map<p,int>dx;
                for(int k=1;k<=n;k++)
                {
                    int x1=str_id[s[k][i]],x2=str_id[s[k][j]];
                    p cnt=make_pair(x1,x2);
                    if(dx.count(cnt))
                    {
                        cout<<"NO"<<endl;
                        cout<<dx[cnt]<<" "<<k<<endl;
                        cout<<i<<" "<<j<<endl;
                        flag=0;
                        goto END;
                    }
                    dx[cnt]=k;
                }
            }
        }
        END:
            if(flag)cout<<"YES"<<endl;
    }
    return 0;
}
